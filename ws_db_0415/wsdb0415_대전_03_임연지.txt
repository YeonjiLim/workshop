SELECT E.ENAME, E.SAL, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;
SELECT E.ENAME, D.DNAME FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.ENAME='KING';
SELECT E.ENAME, E.DEPTNO, D.DNAME, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO;
SELECT CONCAT(E1.ENAME, '의 매니저는 ', E2.ENAME, '이다') '사원과 매니저' FROM EMP E1 JOIN EMP E2 ON E1.MGR=E2.EMPNO;
SELECT E.ENAME, D.DNAME, E.SAL, E.JOB FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.JOB=(SELECT JOB FROM EMP WHERE ENAME='SCOTT');
SELECT EMPNO, ENAME, HIREDATE, SAL FROM EMP E WHERE E.DEPTNO=(SELECT DEPTNO FROM EMP WHERE ENAME='SCOTT');
SELECT E.EMPNO, E.ENAME, E.HIREDATE, D.LOC, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.SAL>=(SELECT AVG(SAL) FROM EMP);
SELECT E.EMPNO, E.ENAME, D.DNAME, D.LOC, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.DEPTNO=30 ORDER BY E.SAL DESC;
SELECT E.EMPNO, E.ENAME, D.DNAME, E.HIREDATE, D.LOC FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.JOB NOT IN (SELECT DISTINCT JOB FROM EMP WHERE DEPTNO=30) AND E.DEPTNO=10;
SELECT E.EMPNO, E.ENAME, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE (E.SAL=(SELECT SAL FROM EMP WHERE ENAME='KING') OR E.SAL=(SELECT SAL FROM EMP WHERE ENAME='JAMES'));
SELECT E.EMPNO, E.ENAME, E.SAL FROM EMP E JOIN DEPT D ON E.DEPTNO=D.DEPTNO WHERE E.SAL > (SELECT MAX(SAL) FROM EMP WHERE DEPTNO=30);
CREATE INDEX EIDX ON EMP(ENAME);
SELECT E.ENAME, E.HIREDATE FROM EMP E WHERE SUBSTR(E.HIREDATE, 1, 4)=(SELECT SUBSTR(HIREDATE, 1, 4) FROM EMP WHERE ENAME='ALLEN');
CREATE VIEW SUMGROUPBYDEPTNO AS SELECT DEPTNO, SUM(SAL) FROM EMP GROUP BY DEPTNO;
