SELECT * FROM emp;
SELECT EMPNO, ENAME, HIREDATE, SAL FROM emp WHERE ENAME LIKE 'K%';
SELECT * FROM emp WHERE substr(HIREDATE, 1, 4)=2000;
SELECT * FROM emp WHERE isnull(COMM)=false;
SELECT ENAME, DEPTNO, SAL FROM emp WHERE DEPTNO=30 and SAL>=1500;
SELECT EMPNO FROM emp WHERE DEPTNO=30 order by EMPNO;
SELECT * FROM emp order by SAL DESC;
SELECT * FROM emp order by DEPTNO, SAL DESC;
SELECT * FROM emp order by DEPTNO DESC, ENAME, SAL DESC;
SELECT ENAME, SAL, COMM, (SAL+COMM) TOTAL FROM emp WHERE isnull(COMM)=false order by TOTAL DESC;
SELECT ENAME, SAL, (SAL*0.13) BONUS, DEPTNO FROM emp WHERE DEPTNO=10;
SELECT ENAME, SAL, ceil(SAL/60) "SAL PER HOUR" FROM emp WHERE DEPTNO=20;
SELECT ENAME, SAL, round(SAL*0.15 ,1) FEE FROM emp WHERE SAL>=1500 and SAL<=3000;
SELECT ENAME, SAL, (SAL-(SAL*0.1)) "NET PAY" FROM emp order by SAL DESC;
SELECT substr(ENAME, 1, 3) FROM emp WHERE length(ENAME)>=6;
SELECT avg(SAL) AVG, max(SAL) MAX, min(SAL) MIN, count(*) CNT FROM emp WHERE DEPTNO=10;
SELECT DEPTNO, JOB, count(*) CNT FROM emp group by DEPTNO, JOB having CNT > 1;
SELECT JOB, count(*) CNT FROM emp group by JOB having CNT>3;
SELECT ENAME, HIREDATE, datediff(now(), HIREDATE) DAYS FROM emp;
SELECT ENAME, timestampdiff(year, HIREDATE, now()) YEARS FROM emp;
